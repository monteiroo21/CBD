// 30. Liste os restaurantes que receberam mais de 5 avaliações e que possuem pelo menos uma avaliação com score de 90.
db.restaurants.aggregate([
    { $unwind: "$grades" },
    {
        $group: {
            _id: "$_id",
            nome: { $first: "$nome" },
            count: { $sum: 1 },
            hasPerfectScore: { $max: { $cond: [ { $eq: ["$requests.score", 90] }, 1, 0 ] } }
        }
    },
    {
        $match: {
            count: { $gt: 5 },
            hasPerfectScore: 1
        }
    },
    {
        $project: {
            _id: 0,
            nome: 1,
            count: 1
        }
    }
])
