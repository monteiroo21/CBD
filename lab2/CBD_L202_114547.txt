// 30. Liste os restaurantes que receberam mais de 5 avaliações e que possuem pelo menos uma avaliação com score de 90.
db.restaurants.aggregate([
  {
    $match: {
      "grades.score": 90
    }
  },
  {
    $addFields: {
      totalGrades: { $size: "$grades" }
    }
  },
  {
    $match: {
      totalGrades: { $gt: 5 }
    }
  },
  {
    $project: {
      nome: 1,
      localidade: 1,
      gastronomia: 1,
      totalGrades: 1,
      grades: 1
    }
  }
])
